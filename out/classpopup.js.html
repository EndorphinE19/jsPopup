/**
 * Создает экземпляр ClassPopup
 * 
 * @constructor
 * @this {ClassPopup}
 */
export default class ClassPopup {
    constructor() {
        /**@property {Object} классы для основного popup окна*/ this._classPopup = {
            mainClassPopup:'popup',
            classClosePage:[
                'popup-close-page',
                {nameForm:'Callback'},
                {animatedClass:'wrapper-popup-content'}
            ]
        },
        /**@property {String} ссылка на текущий сайт*/ this._nameWebsite = location.host,
        this.currentValue = '',
        this.email = '',
        /**@property {RegExp} маска для email*/ this._validateEmailRegExp = /.{1,}@{1,}.{1,}/,
        /**@property {Object} хранит обьекты с сообщениями */ this._errorMessage = {
            /**@property {Object} хранит сообщения для окна настроек */option: {
                optionMessage: 'некорректный адрес, письмо будет отправленно на указанный ранее ящик',
                optionTag: "error-message"
            }
        },
        this.popup = true,
        /**@property {Object} хранит имя класса для элемента с телефоном */ this._classPhone = {
            mainCLassPhone:'phone-popup-module'
        },
        /**@property {String} длительность анимации(для css) */ this._ms = '.3s',
        /**@property {Number} длительность анимации*/ this._msNumber = 300
    }
}
